    Partition
  • ¶

    Divide a stream into two streams, one for which the predicate is true, and one for which the predicate is false.

    partition = (predicate) ->
  (left, right) ->
    (input) ->
      if predicate(input)
        left input
      else
        right input
  • ¶

    Alternate
  • ¶

    Spread a stream among two sinks evenly.

    alternate = (left, right) ->
  leftSide = false
  predicate = ->
    leftSide = !leftSide

  partition(predicate)(left, right)
  • ¶ 
    #! demo
# Alternate streams
[0..9].forEach alternate(
  map((x) -> -x) STDOUT
  STDOUT

    Expose
  • ¶ 
    module.exports = Transducer =
  alternate: alternate
  differentiate: differentiate
  filter: filter
  integrate: integrate
  map: map
  tap: tap
  pollute: ->
    Object.keys(Transducer).forEach (name) ->
      return if name is "pollute"

      global[name] = Transducer[name]
